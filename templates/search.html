Why This is Vulnerable to XSS
Vulnerable Code
# app.py - Search Route
@app.route('/search')
def search():
    query = request.args.get('q', '')
    # ⚠️ NO SANITIZATION!
    return render_template('search.html', 
                         query=query,  # Raw user input
                         results=results)

# search.html - Template
<div>
    Search results for: {{ query|safe }}
    <!-- 🚨 |safe allows HTML/JS injection -->
</div>
Secure Fix
# app.py - Fixed Search Route
from markupsafe import escape

@app.route('/search')
def search():
    query = request.args.get('q', '')
    # ✅ Sanitize input
    safe_query = escape(query)
    return render_template('search.html', 
                         query=safe_query,  # Escaped
                         results=results)

# search.html - Fixed Template
<div>
    Search results for: {{ query }}
    <!-- ✅ Auto-escaped by Jinja2 -->
</div>
Prevention: Always sanitize user input and use auto-escaping in templates!
